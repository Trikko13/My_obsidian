    def __init__(self, start, end):
        self.current = start
        self.end = end

    def __iter__(self):
        return self  # –ò—Ç–µ—Ä–∞—Ç–æ—Ä —Å–∞–º –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç —Å–µ–±—è

    def __next__(self):

            raise StopIteration  # –û—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ–º –∏—Ç–µ—Ä–∞—Ü–∏—é
        value = self.current

        return value

## **2. –ì–µ–Ω–µ—Ä–∞—Ç–æ—Ä—ã (`Generator`)**

üîπ **–ß—Ç–æ —ç—Ç–æ?**

**–ö–∞–∫ —Ä–∞–±–æ—Ç–∞–µ—Ç?**

- `yield` **–∑–∞–º–æ—Ä–∞–∂–∏–≤–∞–µ—Ç** –≤—ã–ø–æ–ª–Ω–µ–Ω–∏–µ —Ñ—É–Ω–∫—Ü–∏–∏ –∏ **–≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –∑–Ω–∞—á–µ–Ω–∏–µ**.
- `next(generator)` **–≤–æ–∑–æ–±–Ω–æ–≤–ª—è–µ—Ç –≤—ã–ø–æ–ª–Ω–µ–Ω–∏–µ** –ø–æ—Å–ª–µ `yield`.

üîπ **–ü—Ä–∏–º–µ—Ä: –ì–µ–Ω–µ—Ä–∞—Ç–æ—Ä –±–µ—Å–∫–æ–Ω–µ—á–Ω–æ–π –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç–∏**


def infinite_numbers():
    num = 1
    while True:
        yield num
        num += 1

gen = infinite_numbers()
print(next(gen))  # 1
print(next(gen))  # 2
print(next(gen))  # 3
